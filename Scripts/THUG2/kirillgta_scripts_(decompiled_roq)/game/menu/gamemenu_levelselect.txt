	:i function $do_globe_state_logic$
		:i switch $globe_state$
			:i case $still$
				:i if NOT  ($globe_new_target_x$ = $globe_target_x$) 
					:i $change$$globe_target_x$ =  ($globe_new_target_x$) 
					:i $change$$globe_state$ = $seeking$
					:i $change$$globe_interpolation$ = %f(0.000000)
					:i $change$$globe_vel$ = %i(0,00000000)
				:i endif
				:i endcase
			case $seeking$
				:i $sin$ ($globe_interpolation$) 
				:i $new_pos$ =  ( ($globe_target_x$ * %GLOBAL%$sin$)  +  ($globe_pos$ *  (%f(1.000000) - %GLOBAL%$sin$) ) ) 
				:i $change$$globe_vel$ =  (%GLOBAL%$new_pos$ - $globe_pos$) 
				:i $change$$globe_pos$ = %GLOBAL%$new_pos$
				:i $change$$globe_interpolation$ =  ($globe_interpolation$ + %f(0.400000)) 
				:i if  ($globe_interpolation$ > %i(90,0000005a)) 
					:i $change$$globe_interpolation$ = %i(90,0000005a)
				:i endif
				:i if  (%GLOBAL%$delta$ < %i(7,00000007)) 
					:i $draw_globe_arrow$
				:i else 
					:i $draw_globe_arrow$$off$
				:i endif
				:i if  (%GLOBAL%$delta$ < %f(0.900000)) 
					:i if NOT  ($globe_state$ = $still$) 
						:i $change$$globe_state$ = $still$
						:i $SpawnScript$$globe_dot_highlight_pulse$
						:i $draw_globe_arrow$
					:i endif
				:i endif
				:i endcase
			case $coasting$
				:i if  ($globe_vel$ < %i(0,00000000)) 
					:i $change$$globe_vel$ =  ($globe_vel$ + %f(0.100000)) 
					:i if  ($globe_vel$ > %i(0,00000000)) 
						:i $change$$globe_state$ = $seeking$
						:i $change$$globe_target_x$ =  ($globe_new_target_x$) 
					:i endif
				:i else 
					:i $change$$globe_vel$ =  ($globe_vel$%f(-0.100000)) 
					:i if  ($globe_vel$ < %i(0,00000000)) 
						:i $change$$globe_vel$ = %i(0,00000000)
						:i if  ($globe_idle_frames$ > %GLOBAL%$idle_threshold$) 
							:i $change$$globe_state$ = $seeking$
							:i $change$$globe_target_x$ =  ($globe_new_target_x$) 
						:i else 
							:i $change$$globe_idle_frames$ =  ($globe_idle_frames$ + %i(1,00000001)) 
						:i endif
					:i endif
				:i endif
				:i $change$$globe_interpolation$ = %f(0.000000)
				:i $change$$globe_pos$ =  ($globe_pos$ + $globe_vel$) 
				:i $change$$globe_vel$ =  ($globe_vel$ * %f(0.900000)) 
				:i endcase
			case $spinning$
				:i $draw_globe_arrow$$off$
				:i if  ($globe_idle_frames$ > %GLOBAL%$idle_threshold$) 
					:i if  ($globe_vel$ < %i(0,00000000)) 
						:i $target_spinning_vel$ = %f(-2.200000)
					:i else 
						:i $target_spinning_vel$ = %f(2.200000)
					:i endif
					:i if  ($globe_vel$ < %GLOBAL%$target_spinning_vel$) 
						:i $change$$globe_vel$ =  ($globe_vel$ + %f(0.100000)) 
						:i if  ($globe_vel$ > %GLOBAL%$target_spinning_vel$) 
							:i $change$$globe_vel$ = %GLOBAL%$target_spinning_vel$
						:i endif
					:i else 
						:i if  ($globe_vel$ > %GLOBAL%$target_spinning_vel$) 
							:i $change$$globe_vel$ =  ($globe_vel$%f(-0.100000)) 
							:i if  ($globe_vel$ < %GLOBAL%$target_spinning_vel$) 
								:i $change$$globe_vel$ = %GLOBAL%$target_spinning_vel$
							:i endif
						:i endif
					:i endif
				:i else 
					:i $change$$globe_idle_frames$ =  ($globe_idle_frames$ + %i(1,00000001)) 
				:i endif
				:i $change$$globe_pos$ =  ($globe_pos$ + $globe_vel$) 
				:i $change$$globe_target_x$ =  ($globe_pos$) 
				:i $change$$globe_new_target_x$ =  ($globe_new_target_x$) 
				:i end_switch
		
	:i endfunction
