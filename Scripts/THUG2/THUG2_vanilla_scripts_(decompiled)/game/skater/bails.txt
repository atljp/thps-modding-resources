:i function $YawBail$
	:i $printf$%s(35,"YawBail ---------------------------")
	:i if $IsSkaterOnVehicle$
		:i $Goto$$vehicle_bail$
	:i endif
	:i $dir$ = $forward$
	:i $GetActualSpin$
	:i if  (%i(0,00000000) > %GLOBAL%$spin$) 
		:i $spin$ =  (%GLOBAL%$spin$ * %i(4294967295,ffffffff)) 
	:i endif
	:i if $LandedFromVert$
		:i $spin$ =  (%GLOBAL%$spin$ + %i(180,000000b4)) 
	:i endif
	:i while
		
		:i if  (%GLOBAL%$spin$ > %i(360,00000168)) 
			:i $spin$ =  (%GLOBAL%$spin$ - %i(360,00000168)) 
		:i else 
			:i continue
			
		:i endif
	:i loop_to 
	:i if  (%i(45,0000002d) > %GLOBAL%$spin$) 
		:i $dir$ = $nose$
	:i else 
		:i if  (%GLOBAL%$spin$ > %i(135,00000087) AND %i(225,000000e1) > %GLOBAL%$spin$) 
			:i $dir$ = $tail$
		:i else 
			:i if $YawingLeft$
				:i $printf$%s(4,"left")
				:i if $Flipped$
					:i $dir$ = $face$
				:i else 
					:i $dir$ = $back$
				:i endif
			:i else 
				:i $printf$%s(5,"right")
				:i if $Flipped$
					:i $dir$ = $back$
				:i else 
					:i $dir$ = $face$
				:i endif
			:i endif
		:i endif
	:i endif
	:i switch %GLOBAL%$dir$
		:i case $nose$
			:i $printf$%s(4,"nose")
			:i $Goto$$Bail_spin_FWD$
			:i endcase
		case $tail$
			:i $printf$%s(4,"tail")
			:i $Goto$$Bail_Spin_BCK$
			:i endcase
		case $back$
			:i $printf$%s(4,"back")
			:i $Goto$$Bail_spin_CW$
			:i endcase
		case $face$
			:i $printf$%s(4,"face")
			:i $Goto$$Bail_spin_CCW$
			:i end_switch
	
:i endfunction

			:i function $PitchBail$
				:i if $IsSkaterOnVehicle$
					:i $Goto$$vehicle_bail$
				:i endif
				:i $tail_script$ = $Bail_Pitch_Tail$
				:i $face_script$ = $Bail_Pitch_Face$
				:i $back_script$ = $Bail_Pitch_Back$
				:i $nose_script$ = $Bail_Pitch_Nose$
				:i if NOT $InSplitscreenGame$
					:i if $GetGlobalFlag$$Flag$ = $TAP_BUTTONS_EXPLAINED$

							:i :s}
							:i $SetGlobalFlag$$Flag$ = $RECOVERY_BUTTONS_EXPLAINED$
						:i endif
					:i endif
				:i endif
				:i $TurnToFaceVelocity$
				:i $GetLastGroundHeading$
				:i $GetHeading$
				:i $dif$ =  (%GLOBAL%$heading$ - %GLOBAL%$last_heading$) 
				:i while
					
					:i if  (%GLOBAL%$dif$ > %i(180,000000b4)) 
						:i $dif$ =  (%i(360,00000168) - %GLOBAL%$dif$) 
					:i else 
						:i if  (%i(4294967116,ffffff4c) > %GLOBAL%$dif$) 
							:i $dif$ =  (%i(360,00000168) + %GLOBAL%$dif$) 
						:i else 
							:i continue
							
						:i endif
					:i endif
				:i loop_to 
				:i $printf$%s(26,"heading=%i last_heading=%j")$i$ = %GLOBAL%$heading$$j$ = %GLOBAL%$last_heading$
				:i $printf$%s(22,"dif =============== %k")$k$ = %GLOBAL%$dif$

					:i if  (%i(0,00000000) > %GLOBAL%$dif$) 
						:i $printf$%s(5,"flip1")
						:i $Flip$
						:i $flipped_skater$ = %i(1,00000001)
					:i endif
				:i else 
					:i if  (%GLOBAL%$dif$ > %i(0,00000000)) 
						:i $printf$%s(5,"flip2")
						:i $Flip$
						:i $flipped_skater$ = %i(1,00000001)
					:i endif
				:i endif

				:i switch %GLOBAL%$pitch_bail_dir$
					:i case $nose$
						:i $pitch_script$ = %GLOBAL%$nose_script$
						:i endcase
					case $tail$
						:i $pitch_script$ = %GLOBAL%$tail_script$
						:i endcase
					case $left$
						:i if $Flipped$
							:i $pitch_script$ = %GLOBAL%$back_script$
						:i else 
							:i $pitch_script$ = %GLOBAL%$face_script$
						:i endif
						:i endcase

							:i $pitch_script$ = %GLOBAL%$face_script$
						:i else 
							:i $pitch_script$ = %GLOBAL%$back_script$
						:i endif
						:i endcase
					default 
						:i $pitch_script$ = %GLOBAL%$face_script$
						:i end_switch
				
				:i if $GotParam$$flipped_skater$
					:i switch %GLOBAL%$pitch_script$
						:i case %GLOBAL%$back_script$
							:i $pitch_script$ = %GLOBAL%$face_script$
							:i endcase
						case %GLOBAL%$face_script$
							:i $pitch_script$ = %GLOBAL%$back_script$
							:i end_switch
					
				:i endif
				:i if $GotParam$$pitch_script$

				:i endif
