:i function $SpecialItem_SyncAnims$$speed$ = %f(1.000000)
	:i $GetTags$
	:i if $GotParam$$Particles$
		:i $Obj_SpawnScript$$Prop_Particles$$id$ = $prop_particles_spawn$$params$ = :s{ isNull :s}
	:i endif
	:i if $GotParam$$Particles2$
		:i $Obj_SpawnScript$$Prop_Particles2$$id$ = $prop_particles2_spawn$$params$ = :s{ isNull :s}
	:i endif
	:i if %GLOBAL%$ParentId$.$Obj_AnimationFlipped$
		:i $Obj_Flip$
	:i endif
	:i if $GotParam$$TurnBoardOff$
		:i %GLOBAL%$ParentId$.$SwitchOffBoard$
	:i endif
	:i while
		
		:i if %GLOBAL%$ParentId$.$AnimEquals$%GLOBAL%$Anim1$
			:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim1$
			:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim1$$NoRestart$%GLOBAL%$cycle$$speed$ = %GLOBAL%$speed$
		:i else 
			:i if $GotParam$$Anim2$
				:i if %GLOBAL%$ParentId$.$AnimEquals$%GLOBAL%$Anim2$
					:i if $GotParam$$wobble$
						:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim2$
						:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim2$$wobble$$wobbleparams$ = $grindwobble_params$$speed$ = %GLOBAL%$speed$
					:i else 
						:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim2$
						:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim2$$cycle$$NoRestart$$speed$ = %GLOBAL%$speed$
					:i endif
				:i else 
					:i if $GotParam$$Anim3$
						:i if %GLOBAL%$ParentId$.$AnimEquals$%GLOBAL%$Anim3$
							:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim3$
							:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim3$$NoRestart$$speed$ = %GLOBAL%$speed$
						:i else 
							:i if NOT $GotParam$$DoNotTurnOff$
								:i continue
								
							:i endif
						:i endif
					:i else 
						:i if NOT $GotParam$$DoNotTurnOff$
							:i continue
							
						:i endif
					:i endif
				:i endif
			:i endif
		:i endif
		:i $wait$%i(1,00000001)
		$game$$frame$
	:i loop_to 
	:i $printf$%s(35,"BREAKING LOOP AND CLEANING UP ITEMS")
	:i if $GotParam$$Particles$
		:i $Obj_KillSpawnedScript$$id$ = $prop_particles_spawn$
		:i $CreateFlexibleParticleSystem$$Name$ = $PizzaBoxFlexSys$$emitscript$ = $emit_skatersplash$$updatescript$ = $skatersplash_update$
	:i endif
	:i if $GotParam$$Particles2$
		:i $Obj_KillSpawnedScript$$id$ = $prop_particles2_spawn$
		:i $DestroyFlexibleParticleSystem$$Name$ = $PizzaBoxFlexSys$
		:i $printf$%s(26,"TURN OFF SPECIAL PARTICLES")
	:i endif
	:i if $GotParam$$TurnBoardOff$
		:i %GLOBAL%$ParentId$.$SwitchOnBoard$
	:i endif
	:i if NOT $GotParam$$DoNotTurnOff$
		:i $printf$%s(1,"3")
		:i %GLOBAL%$ParentId$.$TurnOffSpecialItem$
	:i endif
:i endfunction
:i function $SpecialItem_SyncAnimArray$$speed$ = %f(1.000000)
	:i $GetTags$
	:i $GetArraySize$%GLOBAL%$array$
	:i while
		
		:i $index$ = %i(0,00000000)
		:i while
			
			:i $Anim$ =  (%GLOBAL%$array$:a{%GLOBAL%$index$:a}) 
			:i if %GLOBAL%$ParentId$.$AnimEquals$%GLOBAL%$Anim$
				:i if $GotParam$$wobble$
					:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim$
					:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim$$wobble$$wobbleparams$ = $grindwobble_params$$speed$ = %GLOBAL%$speed$
					:i continue
					
				:i else 
					:i %GLOBAL%$ParentId$.$Obj_GetAnimSpeed$%GLOBAL%$Anim$
					:i $Obj_PlayAnim$$Anim$ = %GLOBAL%$Anim$$cycle$$NoRestart$$speed$ = %GLOBAL%$speed$
					:i continue
					
				:i endif
			:i endif
			:i $index$ =  (%GLOBAL%$index$ + %i(1,00000001)) 
		:i loop_to %GLOBAL%$array_size$
		:i $wait$%i(1,00000001)$gameframe$
	:i loop_to 
:i endfunction
