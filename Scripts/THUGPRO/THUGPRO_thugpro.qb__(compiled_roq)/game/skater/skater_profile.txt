:i function $get_skater_index_from_name$
	:i $GetArraySize$$master_skater_list$
	:i %GLOBAL%$index$ = %i(0,00000000)
	:i while
		
		:i if  (%GLOBAL%$name$ =  ( ($master_skater_list$:a{%GLOBAL%$index$:a}) ->$name$) ) 
			:i return
			$skater_index$ = %GLOBAL%$index$
		:i endif
		:i %GLOBAL%$index$ =  (%GLOBAL%$index$ + %i(1,00000001)) 
	:i loop_to %GLOBAL%$array_size$
	:i $script_assert$%s(56,"Couldn't find profile with name %n in master_skater_list")$n$ = %GLOBAL%$name$
:i endfunction
:i function $get_num_visible_profiles$
	:i $GetArraySize$$master_skater_list$
	:i %GLOBAL%$count$ = %i(0,00000000)
	:i %GLOBAL%$index$ = %i(0,00000000)
	:i while
		
		:i if NOT $StructureContains$$structure$ =  ($master_skater_list$:a{%GLOBAL%$index$:a}) $not_in_frontend$
			:i %GLOBAL%$count$ =  (%GLOBAL%$count$ + %i(1,00000001)) 
		:i endif
		:i %GLOBAL%$index$ =  (%GLOBAL%$index$ + %i(1,00000001)) 
	:i loop_to %GLOBAL%$array_size$
	:i return
	$num_visible_profiles$ = %GLOBAL%$count$
:i endfunction
